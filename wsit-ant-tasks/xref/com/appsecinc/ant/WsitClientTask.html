34  /**
35   * <p>Creates WSIT client configuration files from a WSDL and policy information.</p>
36   * <p>
37   * The task takes the WSDL file specified strips any existing policy information from it
38   * and merges in the policy information provided. The task can take a policy file and
39   * merge it in or it can create a policy for specifying callback handlers automatically.
40   * </p>
41   * <p>
42   * If you specify a policy, the policy will be used. You must also specify a bindingpolicy
43   * to use. The bindingpolicy must be a policy id that exists in the policy document provided.
44   * </p>
45   * <p/>
46   * <p>
47   * If you only need to specify callback handlers, the task can automatically construct
48   * a policy for you. Simply provide any of the callback handler properties
49   * (samlhandler, passwordhandler, usernamehandler) and a policy will be automatically
50   * created and merged into the WSIT configuration file.
51   * </p>
52   * <p/>
53   * <p>
54   * The task will output two files, the first is the client configuration for the service
55   * that the WSDL defines. The second is the main WSIT client configuration file called
56   * wsit-client.xml. If a wsit-client.xml file already exists in the output folder, the
57   * file will be updated.
58   * </p>
59   * <p/>
60   * <p>
61   * To use this task, include a taskdef for the appsecinc-anttasks in your build.xml
62   * </p>
63   * <pre>
64   * <code>
65   * &lt;taskdef resource="com/appsecinc/ant/appsecinc-anttasks.properties" classpath="appsecinc-anttasks.jar" /&gt;
66   * </code>
67   * </pre>
68   * <p>
69   * You can then use the task
70   * </p>
71   * <pre>
72   * <code>
73   * &lt;wsit-client
74   *     outputdir="src/META-INF"
75   *     wsdl="src/wsdl/MyService.wsdl"&gt;
76   *
77   *     &lt;callback
78   *         usernameHandler="com.appsec.UsernameHandler"
79   *         passwordHandler="com.appsec.PasswordHandler" /&gt;
80   *
81   *     &lt;keystore
82   *         aliasSelector="com.appsec.AliasSelector"
83   *         callbackHandler="com.appsec.KeystoreHandler" /&gt;
84   *
85   *     &lt;truststore ..../&gt;
86   *     &lt;certstore ..../&gt;
87   * &lt;/wsit-client&gt;
88   * </code>
89   * </pre>
90   * <p>
91   * This code will produce two files in <code>src/META-INF</code>: <code>MyService.xml</code>
92   * and <code>wsit-client.xml</code>. This configuration tells any MyService clients to
93   * use the specified Username Handler and Password Handler.
94   * </p>
95   * <p/>
96   * <p>
97   * The inner tags (callback, keystore, truststore, and certstore) each allow you to specify
98   * security configuration to support different security scenarios. Not all of the tags are required
99   * for every security scenario. For a good list of security
100  * scenarios, please see http://xwss.java.net/articles/security_config.html
101  * </p>
102  *
103  * @author jhite
104  */

118     public void execute()
119     {
120         try
121         {
122             //Validate properties.
123             checkProperties();
124 
125             WSITBTUtil.doMkDirs(outputdir);
126 
127             //If outputfile is not specified, set it to the name of the wsdl file with a .xml extension.
128             if (outputfile == null)
129             {
130                 outputfile = wsdl.getName();
131                 outputfile = outputfile.substring(0, outputfile.lastIndexOf('.')) + ".xml";
132             }
133 
134             final File wsitOut = new File(outputdir.getAbsoluteFile(), outputfile);
135             FileUtils.getFileUtils().copyFile(wsdl, wsitOut);
136 
137             WsitDocument wsitDoc = WsitDocument.parse(wsitOut, true);
138 
139             if (null != policy)
140             {
141                 log("Using policies from " + policy.getAbsolutePath(), Project.MSG_VERBOSE);
142                 wsitDoc.mergePolicy(policy);
143             }
144             else
145             {
146                 Document defaultPolicy = createDefaultPolicyDocument();
147                 wsitDoc.mergePolicy(defaultPolicy);
148             }
149 
150             wsitDoc.setBindingPolicy(this.bindingpolicy);
151 
152             log("Creating " + wsitOut.getAbsolutePath());
153             wsitDoc.save(wsitOut);
154 
155             //Create a new wsit-client.xml or update it if it exists.
156             final File clientOut = new File(outputdir.getAbsoluteFile(), "wsit-client.xml");
157 
158             final WsitDocument clientDoc;
159             if (clientOut.exists())
160             {
161                 log("Updating WSIT client configuration " + clientOut.getAbsolutePath());
162                 clientDoc = WsitDocument.parse(clientOut);
163             }
164             else
165             {
166                 log("Creating WSIT client configuration " + clientOut.getAbsolutePath());
167                 clientDoc = WsitDocument.newDocument("mainclientconfig");
168             }
169 
170             clientDoc.importWsitDocument(wsitDoc, wsitOut.getName());
171             clientDoc.save(clientOut);
172         }
173         catch (RuntimeException e)
174         {
175             throw new BuildException(e);
176         }
177         catch (Exception e)
178         {
179             throw new BuildException(e);
180         }
181     }

183     private void checkProperties()
184     {
185         if (null == outputdir)
186         {
187             throw new BuildException("The required attribute outputdir must be set");
188         }
189         if (null == wsdl)
190         {
191             throw new BuildException("The required attribute wsdl must be set");
192         }
193         if (!wsdl.exists())
194         {
195             throw new BuildException("The file specified by wsdl attribute must exist. wsdl: " + wsdl.getAbsolutePath());
196         }
197         if (null == bindingpolicy && policy != null)
198         {
199             this.log("A policy is provided but a bindingpolicy is not specified", Project.MSG_WARN);
200         }
201         if (null != policy && null == callback && null == keystore && null == truststore && null == certstore)
202         {
203             this.log("No WSIT Configuration has been specified", Project.MSG_WARN);
204         }
205     }

207     /**
208      * Constructs a wsp:Policy document using the task properties.
209      *
210      * @return
211      * @throws ParserConfigurationException
212      */
213     private Document createDefaultPolicyDocument() throws ParserConfigurationException
214     {
215         DocumentBuilderFactory dbfac = DocumentBuilderFactory.newInstance();
216         dbfac.setNamespaceAware(true);
217         dbfac.setValidating(false);
218         DocumentBuilder docBuilder = dbfac.newDocumentBuilder();
219         Document wsitDoc = docBuilder.newDocument();
220 
221         if (null == bindingpolicy)
222         {
223             bindingpolicy = "Client_BindingPolicy";
224         }
225 
226         Element def = wsitDoc.createElementNS(Namespace.WSDL, "definitions");
227 
228         Element policyElem = wsitDoc.createElementNS(Namespace.WSP, "wsp:Policy");
229         policyElem.setAttributeNS(Namespace.WSU, "wsu:Id", bindingpolicy);
230 
231         Element exactlyOneElem = wsitDoc.createElementNS(Namespace.WSP, "wsp:ExactlyOne");
232         policyElem.appendChild(exactlyOneElem);
233 
234         Element allElem = wsitDoc.createElementNS(Namespace.WSP, "wsp:All");
235         exactlyOneElem.appendChild(allElem);
236 
237         if (null != this.callback)
238         {
239             configureCallbackHandlers(wsitDoc, allElem);
240         }
241 
242         if (null != this.keystore)
243         {
244             configureKeyStore(wsitDoc, allElem);
245         }
246 
247         if (null != this.truststore)
248         {
249             configureTrustStore(wsitDoc, allElem);
250         }
251 
252         if (null != this.certstore)
253         {
254             configureCertStore(wsitDoc, allElem);
255         }
256 
257         if (null != this.secureconversation)
258         {
259             configureSecureConversation(wsitDoc, allElem);
260         }
261 
262         def.appendChild(policyElem);
263         wsitDoc.appendChild(def);
264 
265         return wsitDoc;
266     }

268     private void configureKeyStore(Document wsitDoc, Element parent)
269     {
270         Element ks = wsitDoc.createElementNS(Namespace.SC1, "sc1:KeyStore");
271         ks.setAttributeNS(Namespace.WSPP, "wspp:visibility", "private");
272         parent.appendChild(ks);
273 
274         if (null != keystore.getAlias())
275         {
276             ks.setAttribute("alias", keystore.getAlias());
277         }
278         if (null != keystore.getAliasSelector())
279         {
280             ks.setAttribute("aliasSelector", keystore.getAliasSelector());
281         }
282         if (null != keystore.getCallbackHandler())
283         {
284             ks.setAttribute("callbackHandler", keystore.getCallbackHandler());
285         }
286         if (null != keystore.getKeypass())
287         {
288             ks.setAttribute("keypass", keystore.getKeypass());
289         }
290         if (null != keystore.getLocation())
291         {
292             ks.setAttribute("location", keystore.getLocation());
293         }
294         if (null != keystore.getStorepass())
295         {
296             ks.setAttribute("storepass", keystore.getStorepass());
297         }
298         if (null != keystore.getType())
299         {
300             ks.setAttribute("type", keystore.getType());
301         }
302     }

352     /**
353      * Add callback handler configuration
354      *
355      * @param wsitDoc
356      * @param parent
357      */
358     private void configureCallbackHandlers(Document wsitDoc, Element parent)
359     {
360         Element sc = wsitDoc.createElementNS(Namespace.SC1, "sc1:CallbackHandlerConfiguration");
361         sc.setAttributeNS(Namespace.WSPP, "wspp:visibility", "private");
362         parent.appendChild(sc);
363 
364         if (null != callback.getSamlHandler())
365         {
366             Element handler = wsitDoc.createElementNS(Namespace.SC1, "sc1:CallbackHandler");
367             handler.setAttribute("classname", callback.getSamlHandler());
368             handler.setAttribute("name", "samlHandler");
369             sc.appendChild(handler);
370         }
371         if (null != callback.getUsernameHandler())
372         {
373             Element handler = wsitDoc.createElementNS(Namespace.SC1, "sc1:CallbackHandler");
374             handler.setAttribute("classname", callback.getUsernameHandler());
375             handler.setAttribute("name", "usernameHandler");
376             sc.appendChild(handler);
377         }
378         if (null != callback.getPasswordHandler())
379         {
380             Element handler = wsitDoc.createElementNS(Namespace.SC1, "sc1:CallbackHandler");
381             handler.setAttribute("classname", callback.getPasswordHandler());
382             handler.setAttribute("name", "passwordHandler");
383             sc.appendChild(handler);
384         }
385         if (null != callback.getJmacCallbackHandler())
386         {
387             Element handler = wsitDoc.createElementNS(Namespace.SC1, "sc1:CallbackHandler");
388             handler.setAttribute("classname", callback.getJmacCallbackHandler());
389             handler.setAttribute("name", "jmacCallbackHandler");
390             sc.appendChild(handler);
391         }
392         if (null != callback.getXwssCallbackHandler())
393         {
394             Element handler = wsitDoc.createElementNS(Namespace.SC1, "sc1:CallbackHandler");
395             handler.setAttribute("classname", callback.getXwssCallbackHandler());
396             handler.setAttribute("name", "xwssCallbackHandler");
397             sc.appendChild(handler);
398         }
399     }

401     private void configureSecureConversation(Document wsitDoc, Element parent)
402     {
403         Element sc = wsitDoc.createElementNS(Namespace.SCC, "scc:SCClientConfiguration");
404         sc.setAttributeNS(Namespace.WSPP, "wspp:visibility", "private");
405         sc.setAttribute("renewExpiredSCT", String.valueOf(secureconversation.isRenewExpiredSCT()));
406         parent.appendChild(sc);
407 
408         Element lifetime = wsitDoc.createElementNS(Namespace.SCC, "scc:LifeTime");
409         lifetime.appendChild(wsitDoc.createTextNode(String.valueOf(secureconversation.getLifetime())));
410         sc.appendChild(lifetime);
411     }
